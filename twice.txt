DEBUT
-- Déclaration des variables et initialisations ##
NUM : age;
REAL : salaire, prime;
FIXE NUM : MAX_AGE = 60;
NUM : tableau[5];
NUM : compteur;
TEXT : message;
REAL : age;

EXECUTION {

    -- Calculs arithmétiques ##
    prime <- salaire * 0.1;
    salaire <- salaire + prime;
    affiche("Votre salaire avec prime est : ", salaire);

    -- Conditions imbriquées ##
    SI (age < MAX_AGE) ALORS {
        affiche("Vous pouvez encore travailler.");
        
    } SINON {
		SI (age = MAX_AGE) ALORS {
        	affiche("Vous êtes à l'âge limite pour travailler");
        
   		} SINON {
        	affiche("Vous êtes en âge de retraite.");
        	message <- "Retraite";
    	}
    }

    -- Test des constantes (erreur si on essaie de modifier une constante) ##
    -- Génère une erreur ##
	   MAX_AGE <- 65;
	   
	   
	-- Test de variable non defined ##   
	   x <- x;
	
	--test type incompatible ##   

	   age <- message;
	
    -- Boucles ##  

    TANTQUE (compteur < 3) {
        affiche("Compteur TANTQUE : ", compteur);
        compteur <- compteur + 1;
    }

    -- Tableaux ##
    tableau[0] <- age;
    tableau[1] <- 25;

    affiche("Valeur dans tableau[0] : ", tableau[0]);
    affiche("Valeur dans tableau[1] : ", tableau[1]);

    -- Test dépassement de taille (devrait générer une erreur) ##
    tableau[5] <- 100;
    

    -- Formats d'affichage ##
    affiche("Age : %d, Salaire : %f, Message : %s", age);
}
FIN
